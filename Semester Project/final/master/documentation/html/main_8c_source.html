32 /* SWITCH THE TERMINAL FROM COOKED MODE INTO RAW MODE AN VICE VERSA */
33 void call_termios(int reset)
34 {
35  static struct termios tio, tioOld;
36  tcgetattr(STDIN_FILENO, &tio);
37  if (reset)
38  {
39  tcsetattr(STDIN_FILENO, TCSANOW, &tioOld);
40  }
41  else
42  {
43  tioOld = tio; //backup
44  cfmakeraw(&tio);
45  tio.c_oflag |= OPOST;
46  tcsetattr(STDIN_FILENO, TCSANOW, &tio);
47  }
48 }
